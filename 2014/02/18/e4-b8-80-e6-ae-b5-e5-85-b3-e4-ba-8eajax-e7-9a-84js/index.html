function show_rand_tab(){
var num = Math.floor(Math.random() * 9 + 1);
var tab_index= “marinr_server_cont1_name_” + num;
$(“#”+tab_index).addClass(“on”);
var get_data_id = $(“#”+tab_index).attr(“dataid”);
$.get(“/index.get.data.webservice.php”, { svcid: get_data_id}, function(data){
$(“#”+$(“#”+tab_index).attr(“data”)).html(data);
});
$(“#”+$(“#”+tab_index).attr(“data”)).show();
}
show_rand_tab();
$(“.marinr_server_con_sl > ul > li”).click(function () {
$(“.marinr_server_con_sl > ul > li”).removeClass(“on”);
$(this).addClass(“on”);
var now_obj = $(this);
$(“.marinr_server_cont[data=’marinr_server_cont1’]“).hide();
var has_ajax_data = now_obj.attr( “has_data”);
if(has_ajax_data == “false”){
//Do Ajax.
var data_id = $(this).attr(“dataid”);
$.get(“/index.get.data.webservice.php”, { svcid: data_id}, function(data){
$(“#” + now_obj.attr(“data”)).html(data);
now_obj.attr( “has_data”, “ture” );//To change attr:has_data=ture when get data
});
}else{
//Do nothing.
}
$(“#” + $(this).attr(“data”)).show();

});
