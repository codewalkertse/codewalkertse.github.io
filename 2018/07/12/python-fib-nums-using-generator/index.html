def fib_num_generator():
a, b = 0, 1
while True:
a, b = b, a + b
yield a

def main():
f = fib_num_generator()
fib_list = []
for _ in range(10):
fib_list.append(next(f))
print(fib_list)
if __name__ == “__main__“:
main()
